      def exception_to_css(e, line_offset = 1)
        header = header_string(e, line_offset)

        "/*\n\#{header.gsub(\"*/\", \"*\\\\/\")}\n\nBacktrace:\\n\#{e.backtrace.join(\"\\n\").gsub(\"*/\", \"*\\\\/\")}\n*/\nbody:before {\n  white-space: pre;\n  font-family: monospace;\n  content: \"\#{header.gsub('\"', '\\\"').gsub(\"\\n\", '\\\\A ')}\"; }\n"
      end
