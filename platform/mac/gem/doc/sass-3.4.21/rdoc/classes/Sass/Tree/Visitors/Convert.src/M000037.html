  def visit_children(parent)
    @tabs += 1
    return @format == :sass ? "\n" : " {}\n" if parent.children.empty?

    res = visit_rule_level(parent.children)

    if @format == :sass
      "\n" + res.rstrip + "\n"
    else
      " {\n" + res.rstrip + "\n#{ @tab_chars * (@tabs - 1)}}\n"
    end
  ensure
    @tabs -= 1
  end
